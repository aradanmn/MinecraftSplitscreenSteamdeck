    slot=$1
    event_node=$2
    js_node=$3
    other_controllers=("${@:4}")

    local bwrap_cmd=(bwrap --dev /dev)
    
    # Bind current controller's nodes
    bwrap_cmd+=(--bind "/dev/input/$event_node")
    bwrap_cmd+=(--bind "/dev/js/$js_node")

        bwrap_cmd+=(--bind "/dev/input/$event=/dev/null")
        bwrap_cmd+=(--bind "/dev/js/$js=/dev/null")
    done

    bwrap_cmd+=(

function docked_flow() {
